- [ ] **Step 1: Create the context file**

```tsx
"use client";

import React, { createContext, useContext, useState, ReactNode } from "react";

export interface PortalSettings {
  safetyEnabled: boolean;
  requiredFields: {
    attachments: boolean;
    location: boolean;
    asset: boolean;
    category: boolean;
    priority: boolean;
  };
}

interface Guard1275ContextType {
  settings: PortalSettings;
  toggleSafety: () => void;
  toggleRequiredField: (field: keyof PortalSettings["requiredFields"]) => void;
}

const Guard1275Context = createContext<Guard1275ContextType | undefined>(undefined);

const defaultSettings: PortalSettings = {
  safetyEnabled: true,
  requiredFields: {
    attachments: false,
    location: true,
    asset: false,
    category: true,
    priority: false,
  },
};

export function Guard1275Provider({ children }: { children: ReactNode }) {
  const [settings, setSettings] = useState<PortalSettings>(defaultSettings);

  const toggleSafety = () => {
    setSettings((prev) => ({ ...prev, safetyEnabled: !prev.safetyEnabled }));
  };

  const toggleRequiredField = (field: keyof PortalSettings["requiredFields"]) => {
    setSettings((prev) => ({
      ...prev,
      requiredFields: {
        ...prev.requiredFields,
        [field]: !prev.requiredFields[field],
      },
    }));
  };

  return (
    <Guard1275Context.Provider value={{ settings, toggleSafety, toggleRequiredField }}>
      {children}
    </Guard1275Context.Provider>
  );
}

export function useGuard1275() {
  const context = useContext(Guard1275Context);
  if (context === undefined) {
    throw new Error("useGuard1275 must be used within a Guard1275Provider");
  }
  return context;
}
```

"use client";

import React from "react";
import { useGuard1275 } from "@/contexts/Guard1275Context";
import { ShieldAlert, Paperclip, MapPin, Box, Tag, AlertCircle } from "lucide-react";

export function AdminView() {
  const { settings, toggleSafety, toggleRequiredField } = useGuard1275();

  const renderToggle = (
    label: string, 
    description: string, 
    checked: boolean, 
    onChange: () => void, 
    icon: React.ReactNode,
    isRequiredToggle = false
  ) => (
    <div className="flex items-center justify-between p-4 border rounded-lg mb-3 bg-white">
      <div className="flex items-start gap-3">
        <div className="mt-1 text-gray-500">{icon}</div>
        <div>
          <p className="font-medium text-gray-900">{label}</p>
          <p className="text-sm text-gray-500">{description}</p>
        </div>
      </div>
      <div className="flex items-center gap-3">
        {isRequiredToggle && <span className="text-sm text-gray-500">Required</span>}
        <button
          type="button"
          onClick={onChange}
          className={`relative inline-flex h-6 w-11 flex-shrink-0 cursor-pointer rounded-full border-2 border-transparent transition-colors duration-200 ease-in-out focus:outline-none ${
            checked ? "bg-blue-600" : "bg-gray-200"
          }`}
        >
          <span
            className={`pointer-events-none inline-block h-5 w-5 transform rounded-full bg-white shadow ring-0 transition duration-200 ease-in-out ${
              checked ? "translate-x-5" : "translate-x-0"
            }`}
          />
        </button>
      </div>
    </div>
  );

  return (
    <div className="bg-gray-50 p-6 h-full overflow-y-auto border-r border-gray-200">
      <div className="max-w-2xl mx-auto">
        <h2 className="text-2xl font-bold text-gray-900 mb-6">Edit Portal Settings</h2>
        
        <div className="mb-8">
          <h3 className="text-lg font-semibold text-gray-900 mb-4">Safety Module</h3>
          {renderToggle(
            "Allow Safety Event creation",
            "Enable users to report safety events, near-misses, and hazards through this portal.",
            settings.safetyEnabled,
            toggleSafety,
            <ShieldAlert size={20} />
          )}
        </div>

        <div>
          <h3 className="text-lg font-semibold text-gray-900 mb-4">Request Options</h3>
          <p className="text-sm text-gray-500 mb-4">Enable or disable optional fields.</p>
          
          {renderToggle(
            "Attachments",
            "Let users upload up to 5 photos or files.",
            settings.requiredFields.attachments,
            () => toggleRequiredField("attachments"),
            <Paperclip size={20} />,
            true
          )}
          {renderToggle(
            "Location",
            "Let users specify the exact location.",
            settings.requiredFields.location,
            () => toggleRequiredField("location"),
            <MapPin size={20} />,
            true
          )}
          {renderToggle(
            "Asset",
            "Let users specify an asset.",
            settings.requiredFields.asset,
            () => toggleRequiredField("asset"),
            <Box size={20} />,
            true
          )}
          {renderToggle(
            "Category",
            "Let users choose a category.",
            settings.requiredFields.category,
            () => toggleRequiredField("category"),
            <Tag size={20} />,
            true
          )}
          {renderToggle(
            "Priority",
            "Let users select an urgency level.",
            settings.requiredFields.priority,
            () => toggleRequiredField("priority"),
            <AlertCircle size={20} />,
            true
          )}
        </div>
      </div>
    </div>
  );
}

"use client";

import React, { useState } from "react";
import { useGuard1275 } from "@/contexts/Guard1275Context";
import { MapPin, Mic } from "lucide-react";

export function PortalView() {
  const { settings } = useGuard1275();
  const [activeTab, setActiveTab] = useState<"work_order" | "safety_event">("work_order");

  // If safety is disabled, force tab to work_order
  if (!settings.safetyEnabled && activeTab === "safety_event") {
    setActiveTab("work_order");
  }

  const renderField = (label: string, isRequired: boolean, placeholder: string) => (
    <div className="mb-4">
      <label className="block text-sm font-medium text-gray-700 mb-1">
        {label} {isRequired && <span className="text-red-500">*</span>}
      </label>
      <input
        type="text"
        placeholder={placeholder}
        className="w-full border border-gray-300 rounded-md px-3 py-2 text-sm focus:outline-none focus:ring-1 focus:ring-blue-500"
        readOnly
      />
    </div>
  );

  return (
    <div className="bg-gray-100 h-full overflow-y-auto flex flex-col">
      {/* Header Banner */}
      <div className="bg-blue-600 text-white pt-12 pb-24 px-6 text-center">
        <h1 className="text-3xl font-bold mb-2">Let's Report the Issue</h1>
        <p className="text-blue-100 mb-8">Tell us what's happening and we'll guide you through the rest</p>
        
        <div className="max-w-2xl mx-auto bg-blue-700 rounded-lg p-4 flex items-center gap-3 text-left">
          <MapPin className="text-blue-300" />
          <span>You're reporting an issue for <strong>New York</strong></span>
        </div>
      </div>

      {/* Main Content Area */}
      <div className="max-w-2xl mx-auto w-full px-6 -mt-12 pb-12 flex-1">
        <div className="bg-white rounded-xl shadow-lg p-6">
          
          {/* Tabs */}
          <div className="flex gap-2 mb-6 border-b border-gray-200 pb-2">
            <button
              onClick={() => setActiveTab("work_order")}
              className={`px-4 py-2 rounded-md text-sm font-medium transition-colors ${
                activeTab === "work_order" ? "bg-gray-100 text-gray-900" : "text-gray-500 hover:text-gray-700"
              }`}
            >
              Work Order
            </button>
            {settings.safetyEnabled && (
              <button
                onClick={() => setActiveTab("safety_event")}
                className={`px-4 py-2 rounded-md text-sm font-medium transition-colors ${
                  activeTab === "safety_event" ? "bg-gray-100 text-gray-900" : "text-gray-500 hover:text-gray-700"
                }`}
              >
                Safety Event
              </button>
            )}
          </div>

          {/* Tab Content */}
          {activeTab === "work_order" ? (
            <div>
              <div className="bg-blue-50 text-blue-800 p-3 rounded-md text-sm mb-4">
                ⓘ Use Work Orders to request maintenance, repairs, or report broken equipment.
              </div>
              <div className="relative">
                <textarea
                  className="w-full border border-gray-300 rounded-lg p-4 min-h-[120px] focus:outline-none focus:ring-2 focus:ring-blue-500 resize-none"
                  placeholder="Start describing the problem..."
                  readOnly
                />
                <button className="absolute bottom-4 right-4 p-2 bg-gray-100 rounded-full text-blue-600 hover:bg-gray-200">
                  <Mic size={20} />
                </button>
              </div>
            </div>
          ) : (
            <div>
              <div className="bg-yellow-50 text-yellow-800 p-3 rounded-md text-sm mb-6">
                ⓘ Use Safety Events to report accidents, near-misses, or hazardous observations.
              </div>
              
              <h3 className="text-lg font-bold text-gray-900 mb-1">Safety Event Details</h3>
              <p className="text-sm text-gray-500 mb-6">Please provide specific details about the safety event you are reporting.</p>

              {renderField("Title", true, "e.g. Slip on Wet Floor")}
              
              <div className="grid grid-cols-2 gap-4">
                {renderField("Location", settings.requiredFields.location, "New York")}
                {renderField("Asset", settings.requiredFields.asset, "Select Asset")}
              </div>

              <div className="grid grid-cols-2 gap-4">
                {renderField("Category", settings.requiredFields.category, "Select Category")}
                {renderField("Priority / Severity", settings.requiredFields.priority, "Select Priority")}
              </div>

              <div className="mb-4">
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Attachments {settings.requiredFields.attachments && <span className="text-red-500">*</span>}
                </label>
                <div className="border-2 border-dashed border-gray-300 rounded-lg p-8 text-center text-gray-500">
                  Click to upload or drag and drop
                </div>
              </div>

              <div className="mt-8 flex justify-end">
                <button className="bg-blue-600 text-white px-6 py-2 rounded-lg font-medium hover:bg-blue-700">
                  Submit Request
                </button>
              </div>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
